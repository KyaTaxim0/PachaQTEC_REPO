->SWITCH


let typeCard = 'asa';

switch(typeCard){
    case 'Debit Card':
        console.log('Esta es una tarjeta de debito');
        break;
    case 'Credit Card':
        console.log('Esta es una tarjeta de credito');
        break;
    default:
        console.log('No tienes tarjeta');    
}
